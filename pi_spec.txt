#!/bin/bash
# Raspberry Pi Full Environment Info Script
# Saves output to pi_specs_full.txt

OUTPUT="pi_specs_full.txt"

{
echo "==============================="
echo " ðŸ“Œ Raspberry Pi Experiment Environment "
echo "==============================="

echo -e "\n--- CPU INFO ---"
cat /proc/cpuinfo | grep -E "model name|Hardware|Revision|Serial"

echo -e "\n--- MEMORY INFO ---"
free -h
cat /proc/meminfo | grep MemTotal

echo -e "\n--- DISK INFO ---"
df -h /

echo -e "\n--- OS INFO ---"
cat /etc/os-release
uname -a

echo -e "\n--- GPU + ARM MEMORY ---"
vcgencmd get_mem gpu
vcgencmd get_mem arm

echo -e "\n--- TEMPERATURE ---"
vcgencmd measure_temp

echo -e "\n--- CLOCK SPEEDS ---"
vcgencmd measure_clock arm
vcgencmd measure_clock core

echo -e "\n==============================="
echo " ðŸ“Œ SOFTWARE / TECH STACK INFO "
echo "==============================="

echo -e "\n--- PYTHON VERSION ---"
python3 --version

echo -e "\n--- PIP PACKAGES (main ones) ---"
pip list | grep -E "onnxruntime|numpy|opencv"

echo -e "\n--- VIRTUAL ENV LOCATION ---"
echo $VIRTUAL_ENV

} > "$OUTPUT"

echo "âœ… Full environment info saved to $OUTPUT"
